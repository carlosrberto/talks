Variables & Constants


function doStuff() {
    var arr = [1, 2];
    if(arr.length == 2) {
        var arrLength = arr.length;
    }
    console.log(arrLength);
}

doStuff(); // 2
                        

function doStuff() {
    let arr = [1, 2];
    if(arr.length == 2) {
        let arrLength = arr.length;
    }
    console.log(arrLength);
}

doStuff(); // Uncaught ReferenceError: arrLength is not defined
                        

(function() {
    var age = 29;
})();
console.log(age); // Uncaught ReferenceError: age is not defined
                        

{
    let age = 29;
}
console.log(age); // Uncaught ReferenceError: age is not defined

Arrow Functions


function isFirstLetterUpperCase(word) {
    return (/^[A-Z]/).test(word);
}
                        

let isFirstLetterUpperCase = function(word) {
    return (/^[A-Z]/).test(word);
}
                        

let isFirstLetterUpperCase = (word) => {
    return (/^[A-Z]/).test(word);
}
                        

let isFirstLetterUpperCase = word => (/^[A-Z]/).test(word);

Arrow Functions clean syntax

Before


let arr = [1, 2, 3, 4];
arr
.filter(function(i){
    return i%2 === 0;
})
.map(function(i) {
    return i+1;
});
console.log(arr); // [3, 5]

After


let arr = [1, 2, 3, 4];
arr
.filter(i => i%2 === 0)
.map(i => i+1);
console.log(arr); // [3, 5]

Default parameters

ES5


function f(x, y) {
  y = y || 12;
  return x + y;
}
f(3) // 15

ES6


function f(x, y=12) {
  // y is 12 if not passed (or passed as undefined)
  return x + y;
}
f(3) // 15

Rest parameters

ES5


function f(x, y){
    let args = Array.prototype.slice.apply(arguments, [1, arguments.length]);
    console.log(x, y, args);
}
f(1, 2, 3, 4) // 1 2 [3, 4]

ES6


function f(x, y, ...args){
    console.log(x, y, args)
}
f(1, 2, 3, 4) // 1 2 [3, 4]
