Variables & Constants

let is the new var

ES5


var list = [1, 2];
if(list.length == 2) {
    var listLength = list.length;
}

console.log(listLength); // 2

ES6


let list = [1, 2];
if(list.length == 2) {
    let listLength = list.length;
}

console.log(listLength); // Uncaught ReferenceError: arrLength is not defined

ES5


(function() {
    var age = 29;
})();

console.log(age); // Uncaught ReferenceError: age is not defined

ES6


{
    let age = 29;
}

console.log(age); // Uncaught ReferenceError: age is not defined

Arrow Functions


function isFirstLetterUpperCase(word) {
    return (/^[A-Z]/).test(word);
}
                        

let isFirstLetterUpperCase = function(word) {
    return (/^[A-Z]/).test(word);
}
                        

let isFirstLetterUpperCase = (word) => {
    return (/^[A-Z]/).test(word);
}
                        

let isFirstLetterUpperCase = word => (/^[A-Z]/).test(word);

Clean syntax

ES5


let arr = [1, 2, 3, 4];
arr
.filter(function(i){
    return i%2 === 0;
})
.map(function(i) {
    return i+1;
});
console.log(arr); // [3, 5]

ES6


let arr = [1, 2, 3, 4];
arr
.filter(i => i%2 === 0)
.map(i => i+1);
console.log(arr); // [3, 5]

Default parameters

ES5


function f(x, y) {
  y = y || 12;
  return x + y;
}
f(3) // 15

ES6


function f(x, y=12) {
  // y is 12 if not passed (or passed as undefined)
  return x + y;
}
f(3) // 15

Rest parameters

ES5


function f(x, y){
    let args = Array.prototype.slice.apply(arguments, [1, arguments.length]);
    console.log(x, y, args);
}
f(1, 2, 3, 4) // 1 2 [3, 4]

ES6


function f(x, y, ...args){
    console.log(x, y, args)
}
f(1, 2, 3, 4) // 1 2 [3, 4]

Spread Operator


let list = [2,4,6,8];

let fn = function(a, b, c, d) {
    console.log(a, b, c, d);
}

ES5


fn(list[0], list[1], list[2], list[3]);

fn.apply(null, list);

ES6


fn(...list);
                            

let arr1 = [1,2,3];
let arr2 = [4,5,6];

// arr1.push(v1, v2, v3);


arr1.push(...arr2);


                            

let arr1 = [3, 4];
let arr2 = [1, ...arr1, 5, 6, 7]; // [1, 3, 4, 5, 6, 7]

Rest & Spread Properties

Rest properties


let myObj = {
    x: 1,
    y: 2,
    a: 3,
    b: 4
};

let {x, y, ...z} = myObj;

console.log(x); // 1
console.log(y); // 2
console.log(z); // { a: 3, b: 4 }

Spread properties


let n = {x, y, ...z};
console.log(n); // { x: 1, y: 2, a: 3, b: 4 }
